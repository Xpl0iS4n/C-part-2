Exercise 0: ft_putnbr

#include <unistd.h>

void    ft_putchar(char c)
{
        write(1, &c, 1);
}

void    ft_putnbr(int number)
{
        if (number == -2147483648)
        {
                write(1, "-2147483648", 11);
                return ;
        }
        if (number < 0)
        {
                ft_putchar('-');
                number = -number;
        }
        if (number >= 10)
        {
                ft_putnbr(number / 10);
        }
        ft_putchar(number % 10 + '0');
}

Exercise 1: ft_print_comb

#include <unistd.h>

void    ft_print(char st, char nd, char rd)
{
        write(1, &st, 1);
        write(1, &nd, 1);
        write(1, &rd, 1);
        if (!(st == '7' && nd == '8' && rd == '9'))
                write(1, ", ", 2);
}

void    ft_print_comb(void)
{
        char    st;
        char    nd;
        char    rd;

        st = '0';
        while (st <= '7')
        {
                nd = st + 1;
                while (nd <= '8')
                {
                        rd = nd + 1;
                        while (rd <= '9')
                        {
                                ft_print(st, nd, rd);
                                rd++;
                        }
                        nd++;
                }
                st++;
        }
}
